Entità		Attributi							

CLIENTE		UsernameCliente, PasswordCliente, idCliente, idContoCliente, Credito

VENDITORE	UsernameVend, PasswordVend, idVend, idContoVend

SALDO		idConto, Saldo

OGGETTO		NomeOgg, DescrizOgg, PrezOgg, idOgg, quantitaOgg, idVendOgg

Relazioni

Cliente - Oggetto (1 - n)
Venditore - Oggetto (1 - n)
Cliente - Saldo (1 - 1)
Venditore - Saldo (1 - 1)

//creazione tabelle

CREATE TABLE cliente (
idcliente INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
usernamecliente VARCHAR(128),
passwordcliente VARCHAR(128),
idcontocliente INTEGER
)

CREATE TABLE venditore(
idvend GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
idcontovend INTEGER,
usernamevend VARCHAR(128),
passwordvend VARCHAR(128)
)

CREATE TABLE saldo(
idsaldo GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
saldo DOUBLE,
idcliente INTEGER,
FOREIGN KEY idcliente REFERENCES cliente(idcliente)
)

CREATE TABLE oggetto(
idogg INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nomeogg VARCHAR(128),
urlogg VARCHAR(256),
descrizogg VARCHAR(256),
prezogg DOUBLE,
quantitaogg INTEGER,
idvend INTEGER,
FOREING KEY idvend REFERENCES venditore(idvend)
)
